49  // Really hate loading things like this; however, if I used XML It would be basically
50  // uglier than this. So this is the simplist, most eligant way C++ can handle this I guess.
51 
52  // Load plugin.cfg with read only permission
53  QFile cfg(path);
54  if (!cfg.open(QIODevice::ReadOnly | QIODevice::Text))
55  return;
56 
57  // While there are still lines to read
58  while (!cfg.atEnd())
59  {
60  // Read a line
61  QString line = cfg.readLine();
62 
63  // Comments and blank lines.
64  if(line.contains("#-#"))
65  {
66  while(true)
67  {
68  // If it's the end of the multi-line comment. Break from this cyclical hell.
69  if(cfg.readLine().contains("#-#"))
70  break;
71  // If we've reached the end of the file, then the comment was never closed.
72  if(cfg.atEnd())
73  return;
74  }
75  continue;
76  }
77  if(line == "\n" || line.contains("#"))
78  continue;
79 
80  // If it's not blank, split it at the space
81  QStringList parameters = line.split(" ");
82 
83  // Ensure we have at least two parts
84  if(parameters.size() < 2)
85  continue;
86 
87  // Put the properties into the correct locations.
88  if(parameters[0] == "[Format]")
89  {
90  if(parameters[1].remove("\n").toLower() == "data")
91  {
92  Format = PluginFormat::DATA;
93  }
94  else if(parameters[1].remove("\n").toLower() == "archive")
95  {
96  Format = PluginFormat::ARCHIVE;
97  }
98  else
99  {
100  return;
101  }
102  continue;
103  }
104  if(parameters[0] == "[Name]")
105  {
106  Name = parameters[1].remove("\n");
107  continue;
108  }
109  if(parameters[0] == "[Author]")
110  {
111  Author = parameters[1].remove("\n");
112  continue;
113  }
114  if(parameters[0] == "[CLISupport]")
115  {
116  CLISupported = (parameters[1].toLower() == "true\n") ? true : false;
117  continue;
118  }
119  if(parameters[0] == "[Version]")
120  {
121  Version = parameters[1].remove("\n");
122  continue;
123  }
124  if(parameters[0] == "[URL]")
125  {
126  URL = parameters[1].remove("\n");
127  continue;
128  }
129  if(parameters[0] == "[Lib]")
130  {
131  Lib = parameters[1].remove("\n");
132  continue;
133  }
134  if(parameters[0] == "[Setting]")
135  {
136  QString buffer = parameters[1].split("=")[0];
137  Settings[buffer] = parameters[1].split("=")[1].remove("\n");
138  continue;
139  }
140  if(parameters[0] == "[Extension]")
141  {
142  SupportedFormats.append(parameters[1].remove("\n"));
143  }
144  }
